DOT Diagrams

Basic DOT

digraph {
    A -> B
    B -> C
    A -> C
}

DOT with shapes (circle = green, diamond = yellow, box = blue)

digraph {
    node [shape=circle];
    A [label="Circle Node" fillcolor="#62d256" style=filled];
    node [shape=diamond];
    B [label="Diamond" fillcolor="#fdd835" style=filled];
    node [shape=box];
    C [label="Box" fillcolor="#42a5f5" style=filled];
    A -> B -> C
}

DOT with parallelogram

digraph {
    node [shape=parallelogram];
    A [label="Input"];
    node [shape=box];
    B [label="Process"];
    node [shape=hexagon];
    C [label="Decision"];
